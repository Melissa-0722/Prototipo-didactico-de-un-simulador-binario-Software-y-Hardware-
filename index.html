    public static void decimalABinario(Scanner scanner) {
        System.out.print("\n Ingresa un numero *DECIMAL* (maximo 2,147,483,647): ");

        
        if (!scanner.hasNextInt()) {
            System.out.println("\n Debes teclear un numero entero valido (no letras).");
            scanner.next(); 
            return;
        }

        int numeroDecimal = scanner.nextInt();

        if (numeroDecimal < 0) {
            System.out.println("\n solo puedo convertir numeros *positivos* (mayores o iguales a 0).");
            return;
        }

    
        if (numeroDecimal == 0) {
            System.out.println("‚≠ê El numero 0 en Binario es: *0*");
            return;
        }

  
        String resultadoBinario = ""; 
        int auxiliar = numeroDecimal;

            resultadoBinario = residuo + resultadoBinario;

    }
